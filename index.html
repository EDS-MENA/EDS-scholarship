        function calculateDaysLeft(deadline) {

            const today = new Date();

            today.setHours(0, 0, 0, 0);

            const deadlineDate = new Date(deadline + 'T00:00:00');

            const diffTime = deadlineDate - today;

            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));

            return diffDays;

        }


        function getStatus(deadline) {

            const daysLeft = calculateDaysLeft(deadline);

            return daysLeft >= 0 ? "مفتوح" : "مغلق";

        }


        function formatDate(dateStr) {

            const date = new Date(dateStr + 'T00:00:00');

            return date.toLocaleDateString('ar-EG', { year: 'numeric', month: 'long', day: 'numeric' });

        }


        function renderTable(filtered = scholarships) {

            tableWrapper.style.display = 'block';

        function updateCountryFilter() {

            const countries = [...new Set(scholarships.map(s => s.country))];

            const select = document.getElementById('filterCountry');

            const currentValue = select.value;


            const options = select.querySelectorAll('option:not(:first-child)');

            options.forEach(opt => opt.remove());


            countries.sort().forEach(country => {

                const option = document.createElement('option');

                option.value = country;

                option.textContent = country;

                select.appendChild(option);

            });


            select.value = currentValue;

        }


        function filterScholarships() {

            const search = document.getElementById('searchInput').value.toLowerCase();

            const country = document.getElementById('filterCountry').value;

            const level = document.getElementById('filterLevel').value;

            const status = document.getElementById('filterStatus').value;


            const filtered = scholarships.filter(sch => {

                const matchSearch = sch.name.toLowerCase().includes(search) ||

                                  sch.country.toLowerCase().includes(search) ||

                                  (sch.university && sch.university.toLowerCase().includes(search));

                const matchCountry = !country || sch.country === country;

                const matchLevel = !level || sch.levels.includes(level);

                const matchStatus = !status || getStatus(sch.deadline) === status;


                return matchSearch && matchCountry && matchLevel && matchStatus;

            });


            renderTable(filtered);

        }

            filterScholarships();

        }
